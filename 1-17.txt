Даны три целых числа: A, B, C. Проверить истинность высказывания: «Число B находится между числами A и C».

#include <iostream>
using namespace std;
int main(){
	int A, B, C;
	cout << "Введіть число A: ";
	cin >> A;
	cout << "Введіть число B: ";
	cin >> B;
	cout << "Введіть число C: ";
	cin >> C;
	if (B > A&&B < C || B<A&&B>C) {
 cout << "Вірно";
 }
 else {
 cout << "Невірно";
 }
	return 0;
}
